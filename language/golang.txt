var putBufferPool = sync.Pool {
  New: func() interface{} {
    return &pubBuffer{}
  },
}

func newPutBuffer() *putBuffer {
  return putBufferPool.Get().(*putBuffer)
}

func (b *putBuffer) release() {
  *b = pubBuffer{} //make sure b is cleared, and later user could get a
                   //brand new buffer
  putBufferPool.Put(b)
}
